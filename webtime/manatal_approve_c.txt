Function Time2AmPm($PassTime)  {
	$PassHour = ( SubStr($PassTime,0,2) ) ;
	If ( $PassHour > 12 ) {
		 $PassTime = Str_Pad( ($PassHour - 12) ,2,"0",STR_PAD_LEFT ) . ":" . SubStr($PassTime,3) . "PM" ;
	} else {
		$PassTime . "AM" ;
	}
Return $PassTime ;
}
